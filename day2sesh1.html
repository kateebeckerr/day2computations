#using D() find dg/dx given g(z) = 2z^3 - 10.5z^2 +4.1 
dgdz <- expression(2*z^3 - 10.5*z^2 + 4.1)
dg_dz <- D(dgdz, 'z')
dg_dz
2 * (3 * z^2) - 10.5 * (2 * z)
#find T(y)= (2y^3 + 1)^4 - 8y^3
ty <- expression((2 * y^3 +1)^4 - 8*y^3)
dt_dy <- D(ty, 'y')
dt_dy
4 * (2 * (3 * y^2) * (2 * y^3 + 1)^3) - 8 * (3 * y^2)
